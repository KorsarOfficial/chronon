Copy firmware/test1 layout as the base. Replace main.c with a deterministic workload that runs
for at least 50000 ARM cycles. The workload must:

1. Write a known 16-byte ASCII sequence "MERIDIAN_TT_PASS" to UART_DR (0x40011004).
2. Compute a checksum over a small SRAM array (Fibonacci(20) into sram[0..0x80]).
3. Read back UART SR (0x40011000) — under replay this returns the recorded byte stream.
4. Loop arithmetic that exercises the JIT hot path (>1024 iterations of add/sub/mul/and to
   ensure a block reaches HOT_THRESHOLD).
5. Halt with `b .` (infinite loop) once cycle target reached. Cycle counter is monotonic.

main.c skeleton:

```c
#include <stdint.h>

#define UART_DR (*(volatile uint32_t*)0x40011004u)
#define UART_SR (*(volatile uint32_t*)0x40011000u)

static const char tag[] = "MERIDIAN_TT_PASS";

static uint32_t sram_pad[64] __attribute__((section(".sram_data")));

static void uart_putc(char c) {
    /* SR bit 7 (TXE) is always 1 in this emulator's uart model -> write directly */
    UART_DR = (uint32_t)(unsigned char)c;
}

static uint32_t fib(uint32_t n) {
    uint32_t a = 0u, b = 1u;
    for (uint32_t i = 0u; i < n; ++i) { uint32_t t = a + b; a = b; b = t; }
    return a;
}

int main(void) {
    /* tag write loop -> uart */
    for (uint32_t i = 0u; i < (uint32_t)(sizeof tag - 1u); ++i) uart_putc(tag[i]);

    /* fib(20) into sram_pad[0] for verification */
    sram_pad[0] = fib(20u);

    /* arithmetic loop -> JIT hot path */
    uint32_t s = 0u;
    for (uint32_t i = 0u; i < 2048u; ++i) {
        s += i;
        s ^= (i << 1);
        s &= 0xFFFFFFu;
        s -= (i >> 1);
    }
    sram_pad[1] = s;

    /* halt */
    while (1) { __asm__ volatile ("nop"); }
    return 0;
}
```
