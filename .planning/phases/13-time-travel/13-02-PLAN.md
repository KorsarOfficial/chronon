```c
#define SNAP_MAGIC   0x54544B30u  /* "TTK0" little-endian */
#define SNAP_VERSION 1u
#define SRAM_BASE_ADDR 0x20000000u
#define SRAM_SIZE      (256u * 1024u)

typedef struct snap_blob_s {
    u32 magic;
    u32 version;
    u64 cycle;
    cpu_t     cpu;
    systick_t st;
    nvic_t    nvic;
    scb_t     scb;
    mpu_t     mpu;
    dwt_t     dwt;
    stm32_t   stm32;
    eth_t     eth_state;       /* eth.bus is ignored on save, refilled on restore */
    /* uart_state: included only if tools/main.c reveals uart_t carries non-trivial
       state (RX queue, registers). If uart_t is essentially stateless (MMIO sink only)
       this field is omitted and snap_save/restore leaves p->uart untouched. Decision
       deferred to implementation; verify by reading uart.h and tools/main.c. */
    u32 sram_size;             /* always SRAM_SIZE */
    u8  sram[SRAM_SIZE];
    u32 checksum;              /* XOR32 of all preceding bytes */
} snap_blob_t;

bool snap_save   (snap_blob_t* blob, cpu_t* c, bus_t* bus, tt_periph_t* p);
bool snap_restore(const snap_blob_t* blob, cpu_t* c, bus_t* bus, tt_periph_t* p);

/* File round-trip (cross-session) */
bool snap_save_to_file  (const snap_blob_t* blob, const char* path);
bool snap_load_from_file(snap_blob_t* blob, const char* path);

/* Internal helper exposed for tests. */
u32  snap_xor32(const u8* data, u32 n);
```

```c
u32 snap_xor32(const u8* d, u32 n) {
    u32 a = 0u;
    /* word-wise where aligned, then tail */
    u32 i = 0;
    for (; i + 4u <= n; i += 4u) {
        u32 w; memcpy(&w, d + i, 4); a ^= w;
    }
    for (; i < n; ++i) a ^= ((u32)d[i]) << ((i & 3u) * 8u);
    return a;
}

bool snap_save(snap_blob_t* b, cpu_t* c, bus_t* bus, tt_periph_t* p) {
    if (!b || !c || !bus || !p) return false;
    region_t* sr = bus_find_flat(bus, SRAM_BASE_ADDR);
    if (!sr || sr->size < SRAM_SIZE) return false;
    b->magic     = SNAP_MAGIC;
    b->version   = SNAP_VERSION;
    b->cycle     = c->cycles;
    b->cpu       = *c;
    b->st        = *p->st;
    b->nvic      = *p->nv;
    b->scb       = *p->scb;
    b->mpu       = *p->mpu;
    b->dwt       = *p->dwt;
    b->stm32     = *p->stm32;
    b->eth_state = *p->eth;
    b->eth_state.bus = NULL;        /* don't serialize back-pointer */
    /* if uart_state field is present in snap_blob_t, copy *p->uart here */
    b->sram_size = SRAM_SIZE;
    memcpy(b->sram, sr->buf, SRAM_SIZE);
    b->checksum  = 0u;              /* zero before checksum */
    b->checksum  = snap_xor32((const u8*)b, sizeof(*b) - sizeof(u32));
    return true;
}

bool snap_restore(const snap_blob_t* b, cpu_t* c, bus_t* bus, tt_periph_t* p) {
    if (!b || !c || !bus || !p) return false;
    if (b->magic != SNAP_MAGIC || b->version != SNAP_VERSION) return false;
    if (b->sram_size != SRAM_SIZE) return false;
    /* checksum verify: zero out checksum field, recompute */
    snap_blob_t tmp = *b;
    u32 saved = tmp.checksum;
    tmp.checksum = 0u;
    if (snap_xor32((const u8*)&tmp, sizeof(tmp) - sizeof(u32)) != saved) return false;

    region_t* sr = bus_find_flat(bus, SRAM_BASE_ADDR);
    if (!sr) return false;
    *c       = b->cpu;
    *p->st   = b->st;
    *p->nv   = b->nvic;
    *p->scb  = b->scb;
    *p->mpu  = b->mpu;
    *p->dwt  = b->dwt;
    *p->stm32 = b->stm32;
    *p->eth  = b->eth_state;
    p->eth->bus = bus;              /* fix back-pointer */
    /* if uart_state field present: *p->uart = b->uart_state; */
    memcpy(sr->buf, b->sram, SRAM_SIZE);
    /* flush caches that depend on restored state */
    run_dcache_invalidate();
    /* jit_reset_counters needs the live jit_t — caller responsibility via tt_t in 13-04;
       for now we expose a no-arg shim that resets g_jit_for_run if non-NULL: */
    extern jit_t* g_jit_for_tt;     /* set by tt_create in 13-04, NULL until then */
    jit_reset_counters(g_jit_for_tt);
    return true;
}

bool snap_save_to_file(const snap_blob_t* b, const char* path) {
    if (!b || !path) return false;
    FILE* f = fopen(path, "wb");
    if (!f) return false;
    size_t n = fwrite(b, 1, sizeof(*b), f);
    fclose(f);
    return n == sizeof(*b);
}

bool snap_load_from_file(snap_blob_t* b, const char* path) {
    if (!b || !path) return false;
    FILE* f = fopen(path, "rb");
    if (!f) return false;
    size_t n = fread(b, 1, sizeof(*b), f);
    fclose(f);
    if (n != sizeof(*b)) return false;
    if (b->magic != SNAP_MAGIC || b->version != SNAP_VERSION) return false;
    /* checksum verify */
    snap_blob_t tmp = *b;
    u32 saved = tmp.checksum;
    tmp.checksum = 0u;
    return snap_xor32((const u8*)&tmp, sizeof(tmp) - sizeof(u32)) == saved;
}
```

Create tests/test_tt_snapshot.c. Cover three behaviors:

1. Save twice from the same state -> blobs byte-equal.
2. Save -> mutate cpu/sram -> restore -> matches original byte-equal.
3. 100 consecutive restores complete with mean latency < 100ms (TT-04 budget).
4. Save to file -> load from file -> blob equality.
5. Corrupt one byte after save_to_file -> load_from_file returns false.

Skeleton:

```c
#include "test_harness.h"
#include "core/tt.h"
#include "core/cpu.h"
#include "core/bus.h"
#include "core/nvic.h"
#include "core/run.h"
#include "periph/systick.h"
#include "periph/scb.h"
#include "periph/mpu.h"
#include "periph/dwt.h"
#include "periph/stm32.h"
#include "periph/eth.h"
#include "periph/uart.h"
#include <string.h>
#include <stdio.h>
#include <time.h>

static void setup(cpu_t* c, bus_t* bus, tt_periph_t* p,
                  systick_t* st, nvic_t* nv, scb_t* scb,
                  mpu_t* mpu, dwt_t* dwt, stm32_t* stm32, eth_t* eth, uart_t* u) {
    bus_init(bus);
    /* attach SRAM 0x20000000 / 256K and flash 0x08000000 / 1M */
    static u8 ram[256*1024];
    static u8 flash[1024*1024];
    bus_add_flat(bus, "sram",  0x20000000u, sizeof ram,   true);
    bus_add_flat(bus, "flash", 0x08000000u, sizeof flash, false);
    /* register region buffers; bus_init may auto-allocate, follow existing pattern */
    cpu_reset(c, bus);
    systick_init(st);
    /* nvic_init / scb_init / mpu_init / dwt_init / stm32_init / eth_attach / uart_init:
       use the same calls tools/main.c uses (see source for exact names). */
    p->st = st; p->nv = nv; p->scb = scb;
    p->mpu = mpu; p->dwt = dwt; p->stm32 = stm32; p->eth = eth;
    p->uart = u;
}

int main(void) {
    cpu_t c; bus_t bus;
    systick_t st; nvic_t nv; scb_t scb; mpu_t mpu; dwt_t dwt; stm32_t stm32; eth_t eth;
    uart_t u;
    /* zero-init the uart by default; if tools/main.c shows a uart_init() call use it instead */
    memset(&u, 0, sizeof u);
    tt_periph_t p;
    setup(&c, &bus, &p, &st, &nv, &scb, &mpu, &dwt, &stm32, &eth, &u);

    /* mark cpu state recognizable */
    c.r[0] = 0xDEADBEEFu;
    c.r[1] = 0xCAFEBABEu;
    region_t* sr = bus_find_flat(&bus, 0x20000000u);
    sr->buf[0] = 0xAAu; sr->buf[100] = 0xBBu;

    snap_blob_t a, b;
    TEST(snap_save(&a, &c, &bus, &p), "snap_save a");
    TEST(snap_save(&b, &c, &bus, &p), "snap_save b");
    TEST(memcmp(&a, &b, sizeof a) == 0, "two saves not byte-equal");

    /* mutate state */
    c.r[0] = 0xDEADu;
    sr->buf[0] = 0x11u;
    TEST(snap_restore(&a, &c, &bus, &p), "restore a");
    TEST(c.r[0] == 0xDEADBEEFu && c.r[1] == 0xCAFEBABEu, "cpu not restored");
    TEST(sr->buf[0] == 0xAAu && sr->buf[100] == 0xBBu, "sram not restored");
    TEST(eth.bus == &bus, "eth.bus back-pointer not refilled");

    /* TT-04: 100 restores under 100ms mean (very loose; expected ~26us each) */
    clock_t t0 = clock();
    for (int i = 0; i < 100; ++i) snap_restore(&a, &c, &bus, &p);
    clock_t t1 = clock();
    double ms_total = 1000.0 * (double)(t1 - t0) / (double)CLOCKS_PER_SEC;
    double ms_mean  = ms_total / 100.0;
    TEST(ms_mean < 100.0, "snap_restore too slow");

    /* file round-trip */
    const char* path = "snap_rt.bin";
    TEST(snap_save_to_file(&a, path), "save_to_file");
    snap_blob_t loaded;
    TEST(snap_load_from_file(&loaded, path), "load_from_file");
    TEST(memcmp(&a, &loaded, sizeof a) == 0, "file round-trip not byte-equal");

    /* corruption: flip one byte in the middle, expect load to fail */
    {
        FILE* f = fopen(path, "r+b"); TEST(f, "reopen");
        fseek(f, (long)(sizeof(snap_blob_t) / 2), SEEK_SET);
        u8 g; fread(&g, 1, 1, f); g ^= 0x01u;
        fseek(f, (long)(sizeof(snap_blob_t) / 2), SEEK_SET);
        fwrite(&g, 1, 1, f); fclose(f);
        snap_blob_t corrupt;
        TEST(!snap_load_from_file(&corrupt, path), "corrupt load should fail");
    }
    remove(path);

    PASS();
}
```
