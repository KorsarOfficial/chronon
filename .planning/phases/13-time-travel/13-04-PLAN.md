```c
#include "core/tt.h"
#include "core/run.h"
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

/* g_tt and g_replay_mode are defined in 13-01; g_jit_for_tt is defined in 13-02. */

tt_t* tt_create(u32 stride, u32 max_snaps) {
    if (!stride || !max_snaps) return NULL;
    tt_t* tt = (tt_t*)calloc(1, sizeof(tt_t));
    if (!tt) return NULL;
    tt->stride    = stride;
    tt->max_snaps = max_snaps;
    tt->snaps = (snap_blob_t*)calloc(max_snaps, sizeof(snap_blob_t));
    tt->idx   = (snap_entry_t*)calloc(max_snaps, sizeof(snap_entry_t));
    if (!tt->snaps || !tt->idx) { tt_destroy(tt); return NULL; }
    ev_log_init(&tt->log, 16u);
    g_tt = tt;
    return tt;
}

void tt_destroy(tt_t* tt) {
    if (!tt) return;
    if (g_tt == tt) g_tt = NULL;
    free(tt->snaps); free(tt->idx);
    ev_log_free(&tt->log);
    free(tt);
}

void tt_on_cycle(tt_t* tt, cpu_t* c, bus_t* bus, tt_periph_t* p) {
    if (!tt || !c) return;
    if (g_replay_mode) return;                /* never record during replay */
    if ((c->cycles % tt->stride) != 0u) return;
    if (tt->n_snaps >= tt->max_snaps) return; /* TODO: ring eviction in future phase */
    snap_blob_t* b = &tt->snaps[tt->n_snaps];
    if (!snap_save(b, c, bus, p)) return;
    tt->idx[tt->n_snaps].cycle    = c->cycles;
    tt->idx[tt->n_snaps].snap_idx = tt->n_snaps;
    tt->n_snaps++;
}

/* strong override of weak stub from 13-01 */
void tt_record_irq(u64 cyc, u8 irq) {
    if (g_tt && !g_replay_mode) ev_log_append(&g_tt->log, cyc, EVENT_IRQ_INJECT, (u32)irq);
}
/* strong override of weak stub from 13-01 */
void tt_record_uart_rx(u64 cyc, u8 byte) {
    if (g_tt && !g_replay_mode) ev_log_append(&g_tt->log, cyc, EVENT_UART_RX, (u32)byte);
}
```

Step C — tt_rewind + tt_step_back:

```c
static int tt_bsearch_le(const snap_entry_t* a, u32 n, u64 target) {
    int lo = 0, hi = (int)n - 1, best = -1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        if (a[mid].cycle <= target) { best = mid; lo = mid + 1; }
        else hi = mid - 1;
    }
    return best;
}

bool tt_rewind(tt_t* tt, u64 target, cpu_t* c, bus_t* bus, tt_periph_t* p, jit_t* g) {
    if (!tt || !c || !bus || !p) return false;
    int i = tt_bsearch_le(tt->idx, tt->n_snaps, target);
    if (i < 0) return false;
    bool prev_replay = g_replay_mode;
    g_replay_mode = true;
    if (!snap_restore(&tt->snaps[tt->idx[i].snap_idx], c, bus, p)) {
        g_replay_mode = prev_replay; return false;
    }
    if (target > c->cycles) {
        u32 pos = ev_log_seek(&tt->log, c->cycles);
        run_until_cycle(c, bus, target, p->st, p->scb, g, tt->log.buf, tt->log.n, &pos, p);
    }
    g_replay_mode = prev_replay;
    return true;
}

bool tt_step_back(tt_t* tt, u64 n, cpu_t* c, bus_t* bus, tt_periph_t* p, jit_t* g) {
    if (!tt || !c) return false;
    if (n > c->cycles) return false;
    return tt_rewind(tt, c->cycles - n, c, bus, p, g);
}
```

Note: tt_step_back inherits run_until_cycle's whole-instruction granularity. The final
c->cycles after step_back(N) lies in [target, target + 1] where target = original - N
(target is exact when the boundary instruction is 1-cycle, +1 when it is a 2-cycle op).

```c
void tt_diff(const snap_blob_t* a, const snap_blob_t* b, FILE* out) {
    if (!a || !b || !out) return;
    if (a->cycle != b->cycle) fprintf(out, "cycle: %llu -> %llu\n",
                                      (unsigned long long)a->cycle, (unsigned long long)b->cycle);
    for (int i = 0; i < 16; ++i)
        if (a->cpu.r[i] != b->cpu.r[i])
            fprintf(out, "R%d: 0x%08x -> 0x%08x\n", i, a->cpu.r[i], b->cpu.r[i]);
    if (a->cpu.apsr != b->cpu.apsr)
        fprintf(out, "APSR: 0x%08x -> 0x%08x\n", a->cpu.apsr, b->cpu.apsr);
    /* extend with msp, psp, primask, basepri, control as fields exist in cpu_t */
    /* SRAM range-encoded diff */
    u32 sz = a->sram_size < b->sram_size ? a->sram_size : b->sram_size;
    u32 i = 0u;
    while (i < sz) {
        if (a->sram[i] != b->sram[i]) {
            u32 j = i;
            while (j < sz && a->sram[j] != b->sram[j]) j++;
            fprintf(out, "SRAM[0x%08x..0x%08x]: %u bytes differ\n",
                    SRAM_BASE_ADDR + i, SRAM_BASE_ADDR + j - 1u, j - i);
            i = j;
        } else { i++; }
    }
}
```
