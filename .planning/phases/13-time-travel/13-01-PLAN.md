Build the determinism kernel underneath the time-travel layer. Define the public TT API surface
(ev_t, ev_log_t, tt_periph_t, event-type enum) and eliminate the four hidden non-determinism
sources identified by research: jit static counters, missing replay-mode flag on uart, ungated
external nvic_set_pending, and missing bus region lookup.

stamp) are the foundation for snapshot/replay. Without them snapshot+replay diverges.

uart, ev-log-aware nvic, and a two-run byte-equal regression test.

Step C — regression test tests/test_tt_determinism.c:
```c
#include "test_harness.h"
#include "core/cpu.h"
#include "core/bus.h"
#include "core/nvic.h"
#include "core/jit.h"
#include "core/run.h"
#include "core/tt.h"
#include "periph/systick.h"
#include "periph/scb.h"
#include <string.h>

/* Two independent runs of identical firmware blob must produce byte-equal
   cpu_t at the same step. Smoke test for TT-01. Does not yet exercise full
   peripheral replay — that lands in 13-05. */

static void run_one(cpu_t* c, bus_t* bus, jit_t* g, u64 n) {
    bus_init(bus); /* allocate flash + sram + map mmio */
    /* TODO: load same minimal test blob both runs (e.g. firmware/test1) */
    cpu_reset(c, bus);
    systick_t st; systick_init(&st);
    scb_t scb; scb_init(&scb);
    run_steps_full_g(c, bus, n, &st, &scb, g);
}

int main(void) {
    cpu_t a, b; bus_t ba, bb;
    jit_t ga; memset(&ga, 0, sizeof ga);
    jit_t gb; memset(&gb, 0, sizeof gb);
    run_one(&a, &ba, &ga, 10000);
    run_one(&b, &bb, &gb, 10000);
    TEST(memcmp(&a, &b, sizeof a) == 0, "cpu_t diverges between runs");
    /* SRAM regions byte-equal */
    region_t* ra = bus_find_flat(&ba, 0x20000000u);
    region_t* rb = bus_find_flat(&bb, 0x20000000u);
    TEST(ra && rb, "sram region missing");
    TEST(memcmp(ra->buf, rb->buf, ra->size) == 0, "sram diverges between runs");
    /* explicit reset of jit counters between runs leaves equal state */
    jit_reset_counters(&ga);
    jit_reset_counters(&gb);
    TEST(memcmp(ga.counters, gb.counters, sizeof ga.counters) == 0, "jit ctr diverges");
    PASS();
}
```
