```c
u32 ev_log_seek(const ev_log_t* lg, u64 c) {
    if (!lg || lg->n == 0) return 0u;
    u32 lo = 0u, hi = lg->n;
    while (lo < hi) {
        u32 mid = lo + (hi - lo) / 2u;
        if (lg->buf[mid].cycle < c) lo = mid + 1u; else hi = mid;
    }
    return lo;
}
```

```c
void tt_inject_event(cpu_t* c, bus_t* bus, tt_periph_t* p, const ev_t* e) {
    (void)c; (void)bus;
    if (!p || !e) return;
    switch (e->type) {
        case EVENT_UART_RX:
            /* uart_inject_rx exposed by 13-01 in periph/uart.h */
            extern void uart_inject_rx(uart_t* u, u8 byte);
            if (p->uart) uart_inject_rx(p->uart, (u8)(e->payload & 0xFFu));
            break;
        case EVENT_IRQ_INJECT:
            if (p->nv) nvic_set_pending(p->nv, e->payload);
            break;
        case EVENT_ETH_RX:
            /* deferred: payload encodes side-blob index; implementation in future phase */
            break;
        default: break;
    }
}
```

In src/core/run.c:
```c
u64 run_until_cycle(cpu_t* c, bus_t* bus, u64 target,
                    systick_t* st, scb_t* scb, jit_t* g,
                    const ev_t* log, u32 log_n, u32* log_pos,
                    tt_periph_t* p) {
    u64 start = c->cycles;
    while (c->cycles < target) {
        /* drain any events whose cycle has been reached */
        while (log && log_pos && *log_pos < log_n && log[*log_pos].cycle <= c->cycles) {
            tt_inject_event(c, bus, p, &log[*log_pos]);
            (*log_pos)++;
        }
        u64 next_ev = (log && log_pos && *log_pos < log_n) ? log[*log_pos].cycle : target;
        u64 stop    = (next_ev < target) ? next_ev : target;
        u64 gap     = stop - c->cycles;
        if (gap == 0u) break;     /* defensive: should not happen with monotone log */
        run_steps_full_g(c, bus, gap, st, scb, g);
        if (c->cycles == start && gap > 0u) break;  /* watchdog: no forward progress */
        start = c->cycles;
    }
    return c->cycles - start;
}
```

Note: cpu may overshoot the target by up to one ARM cycle since run_steps_full_g advances by
whole instructions (a 2-cycle Thumb op like LDR/STR/MUL/BL straddling the boundary causes
+1). The loop terminates as soon as c->cycles >= target. Document this in tt.h:
"run_until_cycle returns when c->cycles >= target_cycle (whole-instruction granularity;
overshoot up to one ARM cycle)".

Olympiad style: short locals (c, bus, st, scb, g, p, lo, hi, mid, gap, stop, start, next_ev),
no asserts in hot path, no logging.

Implementation in src/core/tt.c:
```c
bool tt_replay(const snap_blob_t* s, const ev_log_t* lg, u64 target,
               cpu_t* c, bus_t* bus, tt_periph_t* p, jit_t* g) {
    if (!s || !c || !bus || !p) return false;
    bool prev_replay = g_replay_mode;
    g_replay_mode = true;
    if (!snap_restore(s, c, bus, p)) { g_replay_mode = prev_replay; return false; }
    if (target < c->cycles) { g_replay_mode = prev_replay; return false; }
    u32 pos = lg ? ev_log_seek(lg, c->cycles) : 0u;
    const ev_t* buf = lg ? lg->buf : NULL;
    u32 n = lg ? lg->n : 0u;
    run_until_cycle(c, bus, target, p->st, p->scb, g, buf, n, &pos, p);
    g_replay_mode = prev_replay;
    return true;
}
```

Note: while g_replay_mode is true, uart_write suppresses TX fputc (per 13-01) and
nvic_set_pending_ext does not record (it would still log but during pure replay this code
path is not reached because event sources during replay come from the log, not external).

```c
   - on IRQ N executes a different ISR that writes a tag to SRAM[0]
   We then build a synthetic ev_log with one EVENT_IRQ_INJECT at cycle 5000
   and call tt_replay twice from the same snap to the same target_cycle.
   Both runs must produce byte-equal cpu_t and SRAM. */

static u8 g_ram[256*1024];
static u8 g_flash[1024*1024];

static void setup_one(cpu_t* c, bus_t* bus, jit_t* g,
                      tt_periph_t* p,
                      systick_t* st, nvic_t* nv, scb_t* scb,
                      mpu_t* mpu, dwt_t* dwt, stm32_t* stm32, eth_t* eth, uart_t* u) {
    bus_init(bus);
    /* Use the same init pattern as tools/main.c. */
    /* Map flash (read-only firmware), sram (RW), peripherals. */
    /* Loaded firmware: tiny thumb blob that loops `b .` and exposes vector to IRQ N. */
    /* See tools/main.c for exact peripheral attach calls. */
    cpu_reset(c, bus);
    memset(g, 0, sizeof *g);
    p->st = st; p->nv = nv; p->scb = scb; p->mpu = mpu; p->dwt = dwt;
    p->stm32 = stm32; p->eth = eth; p->uart = u;

    region_t* r2 = bus_find_flat(&bus2, 0x20000000u);
    TEST(r1 && r2 && memcmp(r1->buf, r2->buf, r1->size) == 0, "sram mismatch across replays");
```
