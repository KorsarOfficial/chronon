<objective>
Close JIT-06 (100M+ IPS) by adding native codegen for the four highest-frequency Cortex-M
ARM instructions absent from the current 52-family native set: OP_PUSH, OP_POP, OP_T32_LDM,
OP_T32_STM. These dominate FreeRTOS function prologues/epilogues and context switches; their
absence forces the entire enclosing TB to interpreter fallback (codegen_emit returns NULL on
first unsupported op, jit.c:38 -> jit_run.c:79-80 walks ins[] one-at-a-time via execute()).
With 14-VERIFICATION.md identifying these three concrete gaps, this single plan addresses the
top-two by ROI: native PUSH/POP/LDM/STM (largest single gain per verifier) and B.cond fast
path (drop pushfq/popfq for ~+5-10% IPS). Jmp-patch direct chaining is deferred — verifier
estimates +10-20% but adds invalidation complexity that risks regressing TT-06 cycle precision.

Purpose: lift measured 56.33M IPS / 54.2ms to 100M+ IPS / <50ms on test7_freertos 5M-cap run.
Math: PUSH/POP/LDM/STM in a typical FreeRTOS hot path (xQueueGenericSend, xTaskIncrementTick,
prvIdleTask, vTaskSwitchContext) account for ~15-25% of executed instructions. Each interpreted
PUSH {r4-r7,lr} is ~5-10x slower than its native equivalent (5 helper-call wrappers vs 5
inline bus_write or 5 inline mov-to-SRAM). Closing this fall-back loop expects ~1.6-1.9x
overall speedup — comfortably crosses the 100M threshold from 56.33M baseline (1.78x = 100.3M).
B.cond fast path adds ~5% on top.

Output: 4 new native opcode families (52 -> 56), B.cond fast path (no pushfq/popfq), unit test
cross-checking native vs interpreter for byte-equal SP+regs+SRAM, tightened bench assertion
(<50ms enforced as hard fail; ROADMAP criterion now CI-gated).
</objective>

Step A — encoding study. Decoder produces (decoder.c:191-201):
- OP_PUSH: reg_list bits 0..7 = R0..R7, bit 14 = LR. PC bit not in T1 PUSH.
- OP_POP:  reg_list bits 0..7 = R0..R7, bit 15 = PC. LR bit not in T1 POP.
Executor (executor.c:526-563): PUSH walks k=0..14 ascending, writing r[k] (or LR for k=14) to
sp+k_off, then sp -= cnt*4 ONCE up front; POP walks k=0..15 ascending, reading sp+k_off into

stores happen (line 530-531: sp = c->r[REG_SP] - cnt*4; c->r[REG_SP] = sp; loop writes to sp,
sp+4, ..., sp+(cnt-1)*4). POP commits SP AFTER all reads (line 554).

Step C — emit_push. The pattern: 4 phases.

```c
/* Native OP_PUSH — bitmask iteration with optional flat-SRAM inline fast path.
   Bitmask convention from decoder.c:194: bits 0..7 = R0..R7, bit 14 = LR.
   ARM semantics (executor.c:526-540):
       cnt = popcount(reg_list & 0x7FFF)            // bits 0..14, exclude bit 15
       sp_new = c->r[REG_SP] - cnt*4
       c->r[REG_SP] = sp_new
       a = sp_new
       for k in 0..14 ascending if (reg_list & (1<<k)):
           val = (k == 14) ? c->r[REG_LR] : c->r[k]
           bus_write(bus, a, 4, val); a += 4
   Codegen layout:
     1. compile-time popcount cnt; emit `mov eax,[r15+CG_R_OFF+13*4]; sub eax,cnt*4;
        mov [r15+CG_R_OFF+13*4], eax`  — SP is in slot 13 (REG_SP)
     2. flat-SRAM probe: if bus_find_flat(b, sram_base) covers sp_new..sp_new+cnt*4 — but
        sp_new is RUNTIME-VARIABLE. We can't statically prove it. Solution: emit a
        runtime check: `cmp eax, [imm32 region_low]; jb .slow; cmp eax, [imm32 region_high]; ja .slow`.
        Cost: ~16 bytes per PUSH for the guard. If guard succeeds: skip helper-calls, do
        `lea r10, [r14_busbuf + (eax - region.base)]` then N inline `mov [r10+k_off], r[k]`.
     **Simplification for Phase 14.7**: bake region.base/region.size and region.buf as
     compile-time imm64s ONLY if the codegen path can rely on the bus layout being stable
     for the lifetime of the JIT cache. Yes — bus regions are added at boot and never
     removed; jit_flush is called by snap_restore which also resets the JIT, so any rebuild
     of bus regions invalidates compiled TBs naturally. Bake `r14_busbuf = (u64)region->buf`
     into the emitted code as imm64.
     Actually simpler still: don't bake the region pointer; on every PUSH, compile-time
     derive `flat_base_imm32` (region->base, e.g. 0x20000000) and `flat_size_imm32`
     (region->size). Runtime: `cmp eax, flat_base_imm32` (use jb / jae for unsigned bounds),
     `cmp eax, flat_base_imm32+flat_size_imm32-cnt*4` to ensure the entire window fits.
     If both pass, compute `r10 = (eax - flat_base_imm32) + r14_busbuf_imm64` ... but
     r14 holds bus_t*, not the buf. **Add a new emitter ld_busbuf_into_r10**: emit
     `mov r10, imm64(region->buf)` — a 10-byte instruction baked at codegen time. This
     does NOT require bus_t at runtime; the buf pointer is captured at codegen time.
     If bus regions could move (e.g. realloc), this would break. Bus regions are calloc'd
     once in bus_add_flat (bus.c) and never moved — safe to bake.

     Pseudocode (n stores; cnt > 0; k_list = sorted ascending list of set bits in reg_list):
     ```
       ld_eax_sp_slot(cg);                     /* eax = c->r[REG_SP] */
       sub_imm(cg, cnt*4);                     /* eax = sp_new */
       st_eax_sp_slot(cg);                     /* c->r[REG_SP] = sp_new */
       /* runtime bounds check vs flat_region (compile-time-known): */
       /* cmp eax, flat_base; jb .slow */
       cmp_eax_imm32(cg, flat_base);           /* 3D imm32 */
       jcc_rel8(cg, 0x72, ?slow_offset);       /* jb */
       cmp_eax_imm32(cg, flat_base + flat_size - cnt*4);
       jcc_rel8(cg, 0x77, ?slow_offset);       /* ja */
       /* fast path: r10 = (u64)region->buf, then for each k: */
       mov_r10_imm64(cg, (u64)region->buf);    /* 49 BA imm64 */
       /* r11 = r10 + (eax - flat_base) — index into buf */
       /* mov r11d, eax (41 89 C3); add r11, r10 (4D 01 D3); sub r11, imm32 flat_base (49 81 EB imm32) */
       mov_r11d_eax_zero_ext(cg);
       add_r11_r10(cg);
       sub_r11_imm32(cg, flat_base);
       /* for each k in ascending order in reg_list: */
       for (k = 0; k <= 14; k++) if (reg_list & (1<<k)) {
           if (k == 14) ld_ecx(cg, REG_LR);    /* ecx = c->r[14]=LR */
           else         ld_ecx(cg, k);          /* ecx = c->r[k] */
           /* mov [r11 + k_off], ecx — k_off counts in 4 byte steps from sp_new */
           mov_r11_off_ecx(cg, k_off);          /* 41 89 8B disp32 */
           k_off += 4;
       }
       jmp_short(cg, fast_done_offset);         /* skip slow */
       /* .slow: helper-call path */
       /* eax still holds sp_new (was saved? — no, sub clobbered it; we need to recompute) */
       /* Re-load sp_new: mov eax, [r15+CG_R_OFF+13*4] */
       ld_eax(cg, REG_SP);                      /* sp_new from memory */
       for (k = 0; k <= 14; k++) if (reg_list & (1<<k)) {
           /* sub_rsp_16 + WIN64 args + bus_write + add_rsp_16 + fault check; reuse emit_store pattern */
           sub_rsp_16(cg);
           mov_rcx_r14(cg);                     /* rcx = bus */
           mov_edx_eax(cg);                     /* rdx = sp_new */
           add_edx_imm(cg, k_off);              /* rdx += k_off; new helper add_edx_imm */
           mov_r8d_imm(cg, 4);
           if (k == 14) mov_r9d_reg(cg, REG_LR);
           else         mov_r9d_reg(cg, k);
           mov_rax_imm64(cg, (u64)bus_write);
           call_rax(cg);
           add_rsp_16(cg);
           test_al_jnz(cg, 3);
           or_bl_1(cg);
       }
       /* .fast_done: */
     ```

   **Bytes per PUSH {r4-r7,lr} (cnt=5, fast):** ~10(SP load) + ~5(sub_imm) + ~7(SP store)
     + ~16(bounds checks) + ~10(mov_r10_imm64) + ~3+3+7(r11 setup) + 5*(7+7)=70(stores)
     + 2(jmp short) = ~133B. Under the 96B/insn budget — tight; raise codegen_emit budget
     from `n*96+64` to `n*160+64` in this plan to accommodate PUSH/POP/LDM/STM.
   **Bytes per PUSH {r4-r7,lr} (cnt=5, slow):** ~10+5+7 + 5*(4+3+2+2+6+5+10+2+2+4+2+3) ≈
     10+5+7 + 5*45 = 247B. Stays under 32-insn block budget.

   The slow path must run when bus=NULL (test contexts) OR when sp falls outside flat. Both
   safe.
   ```

Add the new emitter helpers in codegen.c (above emit_push):

**For the cnt==0 case** (reg_list==0): emit nothing special — the SP arithmetic is a no-op
(sub eax, 0 is fine; emit it anyway for simplicity). Actually if cnt==0 emit_push must not
do the bounds check (sub goes negative). Add early-out: `if (cnt == 0) return;`.

**Backward jump rel8 calculation**: jb/ja relative offsets need a forward distance to .slow
label. Use the **two-pass / fix-up** approach: emit cmps with placeholder rel8=0; record the
codegen offset of each placeholder; emit the fast body; record fast_done position; backpatch
the rel8 placeholders. This is the same forward-patch technique used by emit_b_cond
(codegen.c:497-498 emits disp32=13 hardcoded because layout is known). For PUSH the fast
body length is data-dependent (cnt-driven), so true backpatch is needed.

**Simpler alternative**: emit fast-then-slow with absolute byte-counted offsets. Compute
fast_body_bytes BEFORE emitting any prologue: 10+3+3+7 (r11 setup) + cnt*(7 or 11 for
ld_ecx + 7 for store) + 2 (jmp_short) = 23 + cnt*(14..18) + 2. Pre-compute, then emit cmps
with these offsets. Mathematical, deterministic — no second pass.

Step D — emit_pop. Symmetric but with three differences:
1. SP commit happens AFTER all reads (load all, then sp += cnt*4).
2. Bit 15 = PC; if set, the loaded value goes to [r15+CG_PC_OFF] AND the block terminates
   (no st_pc emitted by codegen_emit since OP_POP is a terminator per jit.c:29). For
   ARMv7-M Cortex-M4, the loaded PC has bit 0 set (Thumb); mask with `& ~1`.
3. Special case: if loaded PC is in the EXC_RETURN range (0xFFFFFFF0..0xFFFFFFFF) and
   cpu->mode == MODE_HANDLER, the executor calls exc_return (executor.c:556-560). For
   native PUSH/POP this is the rare exception-return path. **Decision**: do NOT inline
   exc_return; if loaded PC has top byte 0xFF, set halted=0 / fall back semantics. Simpler:
   detect EXC_RETURN at runtime via `cmp r10d, 0xFFFFFFF0; jae .slow_pc_path`, where
   .slow_pc_path is a `or bl, 1; jmp epilogue` that triggers interpreter re-execution
   of this block (the bl-failure path already exists from 14-02). Interpreter handles
   exc_return correctly. For non-EXC_RETURN PCs, store PC and exit normally.
   **Simpler still**: only the FreeRTOS PendSV ISR pops EXC_RETURN. The PendSV handler is
   short and rarely hot. Mark `OP_POP-with-bit15-set` as native ONLY when the block is
   not entered via EXC_RETURN context. This is hard to detect statically. **Final
   decision**: native POP supports bit 15 = PC for normal returns; if loaded PC top
   byte == 0xFF (EXC_RETURN), trigger fallback via `or bl, 1` (already wired). Cost:
   one extra cmp + jcc per POP-with-PC; negligible.

   ```c
   /* After loading PC into ecx (the last load): */
   /* mov r10d, ecx; and r10d, 0xFFFFFFF0; cmp r10d, 0xFFFFFFF0; je .exc_return_fallback */
   /* alternative: cmp ecx, 0xFFFFFFF0; jae .exc_return_fallback */
   cmp_ecx_imm32(cg, 0xFFFFFFF0u);
   /* jae +X (5B) -> or bl,1; jmp .epilogue (skip PC store) */
   /* On taken: skip the PC commit; or bl,1; jmp short epilogue. On not-taken (jb): commit PC. */
   ```

**Cleaner approach for composite conds — extract NZCV into AL[3:0]:**
```

                      ; actually byte 3 IS exactly NZCV<<4 (bits 7..4); low 4 bits = bits 27..24 of apsr
                      ; but apsr only uses 31..28 in this codebase (DSP Q flag at bit 27 not used)
                      ; so byte 3 == (NZCV << 4) for any APSR we care about
C0 E8 04              ; shr al, 4                      ; al = 0000NZCV (low nibble)
```
After shr, al = 0bNZCV (4 bits). Then:
- EQ:  test al, 0b0100; jnz taken
- NE:  test al, 0b0100; jz  taken
- CS:  test al, 0b0010; jnz taken
- CC:  test al, 0b0010; jz  taken
- MI:  test al, 0b1000; jnz taken
- PL:  test al, 0b1000; jz  taken
- VS:  test al, 0b0001; jnz taken
- VC:  test al, 0b0001; jz  taken
- HI:  test al, 0b0110; cmp al, 0b0010; je  taken    (C=1 AND Z=0 means low 3 bits == 010 ignoring N; mask first then cmp)
       actually: AND al, 0b0110; cmp al, 0b0010; je

       3C 02             ; cmp al, 2
       0F 84 disp32      ; je taken
       ```

- GE:  N==V — extract bits 3 and 0 of al, compare. Use:
       ```
       88 C4             ; mov ah, al           (2B)
       D0 EC             ; shr ah, 1            (... need shr by 3 to align bit 3 to bit 0)
       ```

       ```
       88 C4             ; mov ah, al           (2B)
       C0 EC 03          ; shr ah, 3            (3B: ah's bit 0 = al's bit 3 = N)
       30 E0             ; xor al, ah           (2B: al bit 0 = N XOR V)
       A8 01             ; test al, 1           (2B)
       0F 84 disp32      ; je  taken (GE: N==V means xor==0)  (6B)
       ```
       Total: 2+3+2+2+6 = 15B.
- LT:  same as GE but jne (`0F 85 disp32`).
- GT:  Z==0 && N==V — combine: after the GE-style xor in al[0], also need Z bit clear.
       After `mov ah, al; shr ah, 3; xor al, ah`, al[0] = N^V. Z bit is bit 2 of al
       (because al = ...NZCV originally, after shr 4 we get bNZCV; after xor with shifted,
       al[0] = N^V, al[2] = Z, al[3] = N (unchanged in upper bits — actually xor al, ah
       only flips low bits where ah is set; ah after shr 3 = 0 0 0 N — only ah[0] is set
       to N. So al unchanged in bits 1,2,3 and bit 0 toggled by N. So:
         al[2] = Z (still original)
         al[0] = N XOR V

       test al, 0x05    ; mask bit 2 (Z) and bit 0 (NeqV)  ; A8 05  (2B)
       jz   taken       ; ZF=1 means both Z=0 AND NeqV=0 -> N==V AND Z==0 -> GT (6B)
       ```
       Wait, test al, 0x05 sets ZF=1 iff (al & 0x05) == 0, i.e. al[2]=0 AND al[0]=0
       i.e. Z=0 AND NXorV=0 i.e. Z=0 AND N==V. That's exactly GT.
       So: `mov ah,al; shr ah,3; xor al,ah; test al,0x05; jz taken`.
       Bytes: 2+3+2+2+6 = 15B.
- LE:  Z==1 || N!=V — opposite of GT: `... test al, 0x05; jnz taken`. Same 15B.

static void ld_al_apsr_byte3(codegen_t* cg) {

/* Per ARM cond, emit (test|cmp|xor|...) leaving ZF set "taken=ZF=value", then
   one jcc rel32. The disp32 must skip the fallthrough store-PC (11B) + a jmp short
   over the taken store-PC (2B), landing at the taken store-PC. Thus disp32 = 13.
   The fallthrough/taken stores sit AFTER the jcc, identical to existing emit_b_cond
   layout (codegen.c:498-502). */
static void emit_b_cond_fast(codegen_t* cg, const insn_t* i) {
    if (i->cond == 0xE) { st_pc(cg, i->pc + 4u + i->imm); return; }
    if (i->cond == 0xF) { st_pc(cg, i->pc + i->size);   return; }

    u32 tgt = i->pc + 4u + i->imm;
    u32 fal = i->pc + i->size;

    /* Setup: al = apsr byte 3; if composite, also pre-shift to low nibble */
    ld_al_apsr_byte3(cg);                 /* 7B: mov al, [r15+APSR_OFF+3] */
    bool need_low_nibble = (i->cond >= 0x8);  /* HI..LE need bNZCV in low nibble */
    if (need_low_nibble) shr_al(cg, 4u);  /* 3B: al = 0bNZCV */

    u8 jcc; /* the final jcc opcode to take the branch */
    bool emitted_test = false;

    switch (i->cond) {
        /* simple bit-tests on byte 3 (no shr) */
        case 0x0: test_al_imm8(cg, 0x40); jcc = 0x85; break; /* EQ: jnz on test set ZF=0 -> jne taken */
        case 0x1: test_al_imm8(cg, 0x40); jcc = 0x84; break; /* NE: je if Z bit clear */
        case 0x2: test_al_imm8(cg, 0x20); jcc = 0x85; break; /* CS: jnz if C bit set */
        case 0x3: test_al_imm8(cg, 0x20); jcc = 0x84; break; /* CC: je  if C bit clear */
        case 0x4: test_al_imm8(cg, 0x80); jcc = 0x85; break; /* MI: jnz if N bit set */
        case 0x5: test_al_imm8(cg, 0x80); jcc = 0x84; break; /* PL: je  if N bit clear */
        case 0x6: test_al_imm8(cg, 0x10); jcc = 0x85; break; /* VS: jnz if V bit set */
        case 0x7: test_al_imm8(cg, 0x10); jcc = 0x84; break; /* VC: je  if V bit clear */
        /* (note: cond 0x0..0x7 would also work without shr_al, but we always
           load byte 3 then test the high nibble masks 0x80/0x40/0x20/0x10
           — no need for low-nibble pre-shift; but if we already shr'd, the
           masks shift too: test al,0x04 instead of 0x40. Re-decide:
           DON'T shr for simple conds; only shr for composite. The
           need_low_nibble guard above achieves this — simple conds skip shr. */
        case 0x8: /* HI: C && !Z, on low nibble: bit 1 (C) set, bit 2 (Z) clear */
            and_al_imm8(cg, 0x06); cmp_al_imm8(cg, 0x02); jcc = 0x84; break;
        case 0x9: /* LS: !C || Z */
            and_al_imm8(cg, 0x06); cmp_al_imm8(cg, 0x02); jcc = 0x85; break;
        case 0xA: /* GE: N==V */
            mov_ah_al(cg); shr_ah(cg, 3u); xor_al_ah(cg);
            test_al_imm8(cg, 0x01); jcc = 0x84; break;
        case 0xB: /* LT: N!=V */
            mov_ah_al(cg); shr_ah(cg, 3u); xor_al_ah(cg);
            test_al_imm8(cg, 0x01); jcc = 0x85; break;
        case 0xC: /* GT: !Z && N==V */
            mov_ah_al(cg); shr_ah(cg, 3u); xor_al_ah(cg);
            test_al_imm8(cg, 0x05); jcc = 0x84; break;
        case 0xD: /* LE: Z || N!=V */
            mov_ah_al(cg); shr_ah(cg, 3u); xor_al_ah(cg);
            test_al_imm8(cg, 0x05); jcc = 0x85; break;
        default: /* unreachable: 0xE/0xF handled above */
            return;
    }

    /* jcc rel32 = 13: 11B fall st_pc + 2B jmp short = land at taken st_pc */
    emit_jcc_rel32(cg, jcc, 13u);
    st_pc(cg, fal);                       /* 11B */
    jmp_short(cg, 11u);                   /* 2B: skip taken */
    st_pc(cg, tgt);                       /* 11B */
}

#define SRAM_BASE 0x20000000u
#define SRAM_SIZE 4096u

static void setup(cpu_t* cpu, bus_t* bus) {
    bus_init(bus);
    bus_add_flat(bus, "flash", 0x00000000u, 4096u, false);
    bus_add_flat(bus, "sram",  SRAM_BASE,    SRAM_SIZE, true);
    cpu_reset(cpu, CORE_M4);
    /* Seed regs with distinguishable values */
    for (int k = 0; k < 13; k++) cpu->r[k] = 0xA0000000u | (u32)k;
    cpu->r[REG_LR] = 0xCAFE0E00u;       /* LR */
    cpu->r[REG_SP] = SRAM_BASE + 256u;  /* mid-SRAM */
    cpu->r[REG_PC] = 0x00000000u;
    cpu->halted = false;
}

static void seed_sram(bus_t* bus) {
    /* Pre-seed SRAM with a recognizable pattern so POP/LDM read deterministic values */
    for (u32 a = SRAM_BASE; a < SRAM_BASE + 256u; a += 4u) {
        bus_w32(bus, a, 0xB0000000u | (a - SRAM_BASE));
    }
}

static void snapshot_sram(bus_t* bus, u32* out, u32 n_words) {
    for (u32 k = 0; k < n_words; k++) {
        out[k] = bus_r32(bus, SRAM_BASE + k * 4u);
    }
}

static int diff_state(const cpu_t* a, const cpu_t* b, const u32* sa, const u32* sb, u32 n_words) {
    int errs = 0;
    for (int k = 0; k < 16; k++) {
        if (a->r[k] != b->r[k]) {
            fprintf(stderr, "  r[%d]: native=%08x interp=%08x\n", k, a->r[k], b->r[k]);
            errs++;
        }
    }
    for (u32 k = 0; k < n_words; k++) {
        if (sa[k] != sb[k]) {
            fprintf(stderr, "  sram[%u]: native=%08x interp=%08x\n", k, sa[k], sb[k]);
            errs++;
        }
    }
    return errs;
}

static int run_one(insn_t ins, const char* label) {
    cpu_t cpu_n, cpu_i;
    bus_t bus_n, bus_i;

    setup(&cpu_n, &bus_n); seed_sram(&bus_n);
    setup(&cpu_i, &bus_i); seed_sram(&bus_i);

    /* Native */
    memset(&s_jit, 0, sizeof s_jit);
    jit_init(&s_jit);
    insn_t buf[1] = { ins };
    cg_thunk_t fn = codegen_emit(&s_jit.cg, &bus_n, buf, 1u);
    if (!fn) { fprintf(stderr, "%s: codegen_emit returned NULL\n", label); return 1; }
    if (!fn(&cpu_n, &bus_n)) { fprintf(stderr, "%s: thunk returned false\n", label); return 1; }

    /* Interpreter */
    cpu_i.r[REG_PC] = ins.pc;
    if (!execute(&cpu_i, &bus_i, &ins)) {
        fprintf(stderr, "%s: execute returned false\n", label); return 1;
    }
    cpu_i.r[REG_PC] = ins.pc + ins.size;

    /* Diff */
    u32 sram_n[64], sram_i[64];
    snapshot_sram(&bus_n, sram_n, 64u);
    snapshot_sram(&bus_i, sram_i, 64u);
    int errs = diff_state(&cpu_n, &cpu_i, sram_n, sram_i, 64u);
    if (errs) fprintf(stderr, "%s: %d mismatch(es)\n", label, errs);
    return errs;
}
